-- 不能在回调函数、PostInit 等内部使用。
-- 它读取的是玩家在游戏内"Mod 配置"面板设置的选项值。

local MY_NUMBER = GetModConfigData("MY_NUMBER_OPTION")  -- 返回数字/字符串/布尔
local MY_BOOL   = GetModConfigData("MY_BOOL_OPTION")
local MY_STRING = GetModConfigData("MY_STRING_OPTION")
-- 如果配置项的 name 在 modinfo 中不存在，返回 nil

-- 典型用法：把配置值用于后续逻辑

-- 注意：如果要配合 PostInit，需把值存在局部变量中，再在回调里引用：
local damage_mult = MY_NUMBER or 1.0  -- 兜底默认值
AddPrefabPostInit("axe", function(inst)
    if inst.components.weapon then
        inst.components.weapon:SetDamage(
            TUNING.AXE_DAMAGE * damage_mult
        )
    end
end)
