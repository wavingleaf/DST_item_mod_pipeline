# 字符串与本地化

```lua
-- ============================================================
-- 字符串（STRINGS）重载 — 修改物品名称/描述/检查文本
-- 本地化 — 多语言翻译
-- ============================================================

-- ──────────────────────────────
-- STRINGS 结构速览
-- ──────────────────────────────

-- STRINGS.NAMES                                    — 物品/生物名称
-- STRINGS.RECIPE_DESC                               — 配方简介（鼠标悬停配方时）
-- STRINGS.CHARACTERS.GENERIC.DESCRIBE               — 通用角色的检查文本
-- STRINGS.CHARACTERS.WILSON.DESCRIBE                — Wilson 专属检查文本
-- STRINGS.CHARACTERS.WILLOW.DESCRIBE                — Willow 专属检查文本
-- STRINGS.CHARACTERS.WOLFGANG.DESCRIBE              — Wolfgang 专属检查文本
-- ... 每个角色都有独立的 DESCRIBE 表

-- ──────────────────────────────
-- 修改原版物品的文本
-- ──────────────────────────────

-- 改名
-- STRINGS.NAMES.AXE = "超级斧头"

-- 改检查文本（通用，所有角色默认使用）
-- STRINGS.CHARACTERS.GENERIC.DESCRIBE.AXE = "这斧子变得好厉害！"

-- 改检查文本（角色专属，覆盖通用文本）
-- STRINGS.CHARACTERS.WILSON.DESCRIBE.AXE = "我可以砍倒一切！"
-- STRINGS.CHARACTERS.WILLOW.DESCRIBE.AXE = "带着它让我感觉很安全。"

-- ──────────────────────────────
-- 给自定义物品添加文本
-- ──────────────────────────────

-- 名称
-- STRINGS.NAMES.MY_CUSTOM_SWORD = "秘银宝剑"

-- 配方简介（制作栏悬停时显示）
-- STRINGS.RECIPE_DESC.MY_CUSTOM_SWORD = "一把蕴含魔力的武器"

-- 通用检查文本
-- STRINGS.CHARACTERS.GENERIC.DESCRIBE.MY_CUSTOM_SWORD = "看起来很锋利！"

-- ──────────────────────────────
-- STRINGS 设置的位置
-- ──────────────────────────────

-- 方式 A：直接在 modmain.lua 顶层设置（简单 mod 推荐）
--   全局变量赋值即可，游戏启动后立即生效。

-- 方式 B：放在独立的 lua 文件中，通过 modimport 引入（中大型 mod 推荐）
--   modimport("scripts/strings.lua")

-- 方式 C：使用 .po 文件（多语言 mod 最佳实践，见下方本地化部分）

-- ──────────────────────────────
-- 修改制作栏分页名称
-- ──────────────────────────────

-- STRINGS.TABS.MY_CUSTOM_TAB = "我的分页"

-- ──────────────────────────────
-- 修改状态/动作文本
-- ──────────────────────────────

-- STRINGS.ACTIONS  — 动作名称
-- STRINGS.UI       — 界面文本
-- STRINGS.NAMES    — 名字

-- ──────────────────────────────
-- ──────────────────────────────
-- 本地化（多语言翻译）
-- ──────────────────────────────
-- ──────────────────────────────

-- 如果 Mod 需要支持多语言，推荐使用 .po 文件而非直接在 lua 中赋值。
-- 文件路径：scripts/languages/ 或 scripts/localization/

-- 创建一个 .pot 模板文件（示例）:
--   msgid "秘银宝剑"
--   msgstr ""                  ← 留空，由各语言翻译填充
--
--   msgid "一把蕴含魔力的武器"
--   msgstr ""

-- 创建中文翻译 .po 文件（scripts/languages/chinese.po）:
--   msgid "Mythril Sword"
--   msgstr "秘银宝剑"
--
--   msgid "A weapon imbued with magic."
--   msgstr "一把蕴含魔力的武器"

-- 创建英文翻译 .po 文件（scripts/languages/english.po）:
--   msgid "Mythril Sword"
--   msgstr "Mythril Sword"   ← 原文即英文
--
--   msgid "A weapon imbued with magic."
--   msgstr "A weapon imbued with magic."

-- 然后在 modmain.lua 中使用:
-- STRINGS.NAMES.MY_CUSTOM_SWORD = GetString(
--     GetModConfigData("Language"),  -- GetModConfigData/GetString 是 Klei 注入 modmain 的函数，不在 GLOBAL 表里，不要加 GLOBAL. 前缀
--     "Mythril Sword"
-- )

-- 实际上 Klei 的本地化系统更自动——只要 .po 文件存在，
-- 游戏会根据玩家的语言设置自动加载对应翻译。
-- 具体实现建议参考官方文档和社区模板。
```
