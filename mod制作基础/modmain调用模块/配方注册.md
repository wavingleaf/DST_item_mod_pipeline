# 配方注册

```lua
-- ============================================================
-- 配方注册（Recipes）
-- 位置：modmain.lua 顶层，或通过 modimport 引入的独立文件
-- ============================================================

-- 前置：确保已在 modmain.lua 顶部做了 GLOBAL 环境注入
-- GLOBAL.setmetatable(env, { __index = function(t,k) return GLOBAL.rawget(GLOBAL,k) end })

-- ──────────────────────────────
-- 常用常量引用
-- ──────────────────────────────

-- 以下 Ingredient、RECIPETABS、TECH 等由 modmain 的 metatable 从 GLOBAL 代理，
-- 直接使用即可，无需 local 别名，也无需写 GLOBAL. 前缀

-- RECIPETABS 常用值（制作栏分页）:
--   RECIPETABS.TOWN       -- 建筑
--   RECIPETABS.FARM       -- 食物
--   RECIPETABS.SURVIVAL   -- 生存
--   RECIPETABS.WAR        -- 战斗
--   RECIPETABS.TOOLS      -- 工具
--   RECIPETABS.SCIENCE    -- 科学
--   RECIPETABS.MAGIC      -- 魔法
--   RECIPETABS.DRESS      -- 服饰
--   RECIPETABS.ANCIENT    -- 远古
--   RECIPETABS.CARTOGRAPHY-- 制图
--   RECIPETABS.SCULPTING  -- 雕塑

-- TECH 常用值（科技等级）:
--   TECH.SCIENCE_ZERO     -- 无需科技（随时可做）
--   TECH.SCIENCE_ONE      -- 科学机器
--   TECH.SCIENCE_TWO      -- 炼金引擎
--   TECH.MAGIC_ONE        -- 灵子分解器
--   TECH.MAGIC_TWO        -- 暗影操控器
--   TECH.MAGIC_THREE      -- 远古伪科学站
--   TECH.ANCIENT_TWO      -- 远古（破损）
--   TECH.ANCIENT_FOUR     -- 远古（完整）

-- ──────────────────────────────
-- 方式一：AddRecipe（旧版 API）
-- 兼容性好，社区教程广泛使用，至今仍可用
-- ──────────────────────────────

-- AddRecipe(name, ingredients, tab, level, ...)
-- 参数说明（按顺序）:
--   1. name          (string)  配方产物的 Prefab 名
--   2. ingredients   (table)   材料表 { Ingredient(材料名, 数量), ... }
--   3. tab           (enum)    制作栏分页（RECIPETABS.WAR 等）
--   4. level         (enum)    所需科技等级（TECH.SCIENCE_ONE 等）
--   5. min_spacing   (nil/num) 可选，物品栏最小间距，nil 为默认
--   6. nounlock      (nil/bool)可选，是否无需解锁（nil = 需要解锁）
--   7. numtogive     (nil/num) 可选，制作一次给几个
--   8. builder_tag   (nil/str) 可选，限制只有特定角色能做
--   9. atlas         (nil/str) 可选，制作栏内图集的 xml 路径
--  10. image         (nil/str) 可选，制作栏内图标的 tex 文件名
--  11. product       (nil/str) 可选，产物 Prefab（默认与 name 相同）
--  12. character   (nil/str)   可选，仅指定角色可做
--  13. filters     (nil/table) 可选，筛选项
--  14. description (nil/str)   可选，配方描述

-- 示例：一把简单的自定义剑
local my_sword = AddRecipe(
    "my_custom_sword",                              -- [必需] 产物 Prefab 名
    { Ingredient("twigs", 3), Ingredient("flint", 2) }, -- [必需] 材料
    RECIPETABS.WAR,                                 -- [必需] 制作栏分页
    TECH.SCIENCE_ONE                                -- [必需] 科技等级
    -- 后续参数可省略（nil），以下为可选：
    -- nil,    -- min_spacing
    -- nil,    -- nounlock
    -- nil,    -- numtogive
    -- nil,    -- builder_tag
    -- "images/inventoryimages/my_sword.xml",  -- atlas（图集 xml）
    -- "my_sword.tex",                          -- image（图标 tex）
)

-- ──────────────────────────────
-- 方式二：AddRecipe2（新版 API）
-- 参数以 table 形式传入，更清晰
-- ──────────────────────────────

-- AddRecipe2(name, ingredients, tech_level, config_table)
-- config_table 常用键:
--   atlas       -- 制作栏内图集 xml 路径
--   image       -- 制作栏内图标 tex 文件名
--   product     -- 产物 Prefab（默认与 name 相同）
--   numtogive   -- 制作一次给几个
--   builder_tag -- 限制角色
--   nounlock    -- 是否不需要解锁
--   min_spacing -- 最小间距
--   placer      -- 放置预览 Prefab（建筑类）

-- 2. 需要 .xml（图集坐标）+ .tex（纹理）两个文件

-- 4. 图标推荐尺寸：在 64x64 画布内绘制
```
